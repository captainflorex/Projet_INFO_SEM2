22int menu_mettre_a_jour(EtatMenu *em) {
23
24 /* Anti-répétition : on attend que la touche soit relâchée */

28
29 /* Flèche BAS */
30 if (key[KEY_DOWN] && touche_bas_relachee) {
31 em->element_selectionne = (em->element_selectionne + 1) % NB_ELEMENTS_MENU;
32 touche_bas_relachee = 0;
33 }
34 if (!key[KEY_DOWN]) touche_bas_relachee = 1;
35
36 /* Flèche HAUT */
37 if (key[KEY_UP] && touche_haut_relachee) {
38 em->element_selectionne = (em->element_selectionne + NB_ELEMENTS_MENU - 1)
39 % NB_ELEMENTS_MENU;
40 touche_haut_relachee = 0;
41 }
42 if (!key[KEY_UP]) touche_haut_relachee = 1;
43
44 /* ENTRÉE — validation */
45 if (key[KEY_ENTER] && touche_entree_relachee) {
46 touche_entree_relachee = 0;
47 if (em->element_selectionne == MENU_REGLES) {
48 em->afficher_regles = 1; /* active l'écran des règles */
49 return -1; /* pas de changement d'état */
50 }
51 return em->element_selectionne;
52 }
53
54 if (!key[KEY_ENTER]) touche_entree_relachee = 1;
55
56 /* Retour depuis les règles */
57 if (em->afficher_regles && key[KEY_ESC]) {
58 em->afficher_regles = 0;
59 }
60
61 return -1;
62}

65void menu_saisir_pseudo(EtatMenu *em) {
66 clear_keybuf();
67 int len = strlen(em->pseudo_saisi);
68
69 while (1) {
70 /* Affichage géré dans main.c, ici on lit juste le clavier */
71 if (keypressed()) {
72 int c = readkey();
73 int ascii = c & 0xFF;
74 int scan = (c >> 8) & 0xFF;
75
76 if (scan == KEY_ENTER && len > 0) {
77 break; /* Pseudo validé */
78 } else if (scan == KEY_BACKSPACE && len > 0) {
79 em->pseudo_saisi[--len] = '\0';
80 } else if (ascii >= 32 && ascii < 127 && len < PSEUDO_LEN - 1) {
81 em->pseudo_saisi[len++] = (char)ascii;
82 em->pseudo_saisi[len] = '\0';
83 }
84 }
85 }
